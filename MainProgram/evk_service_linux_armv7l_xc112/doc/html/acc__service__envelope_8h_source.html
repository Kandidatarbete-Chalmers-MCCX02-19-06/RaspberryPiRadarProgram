145 /**
146  * @brief Get running average factor
147  *
148  * The running average factor is the factor of which the most recent sweep is weighed against previous sweeps.
149  * Valid range is between 0.0 and 1.0 where 0.0 means that no history is weighed in, i.e filtering is effectively disabled.
150  * A factor of 1.0 means that the most recent sweep has no effect on the result,
151  * which will result in that the first sweep is forever received as the result.
152  * The filtering is coherent and is done on complex valued IQ data before conversion to envelope data.
153  *
154  * @param[in] service_configuration The configuration to get the running average factor for
155  * @return Running average factor
156  */
157 extern float acc_service_envelope_running_average_factor_get(acc_service_configuration_t service_configuration);
