203 /**
204  * @brief Get running average factor
205  *
206  * The running average factor is the factor of which the most recent sweep is weighed against previous sweeps.
207  * Valid range is between 0.0 and 1.0 where 0.0 means that no history is weighed in, i.e filtering is effectively disabled.
208  * A factor of 1.0 means that the most recent sweep has no effect on the result,
209  * which will result in that the first sweep is forever received as the result.
210  * The filtering is coherent and is done on complex valued IQ data before conversion to envelope data.
211  *
212  * @param[in] configuration The configuration to get the running average factor for
213  * @return Running average factor
214  */
215 extern float acc_detector_distance_peak_running_average_factor_get(acc_detector_distance_peak_configuration_t configuration);
