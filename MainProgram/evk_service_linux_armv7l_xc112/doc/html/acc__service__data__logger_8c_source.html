97 int main(int argc, char *argv[])
98 {
99  input_t input;
100 
101  initialize_input(&input);
102 
103  signal(SIGINT, interrupt_handler);
104 
105  acc_log_set_level(ACC_LOG_LEVEL_FATAL, NULL);
106 
107  if (!acc_rss_activate()) {
108  return EXIT_FAILURE;
109  }
110 
111  if (parse_options(argc, argv, &input) != ACC_STATUS_SUCCESS) {
112  if (input.file_path != NULL) {
113  acc_os_mem_free(input.file_path);
114  }
115  return EXIT_FAILURE;
116  }
117 
118  acc_service_status_t service_status;
119 
120  switch (input.service_type) {
121  case POWER_BIN:
122  {
123  acc_service_configuration_t power_bin_configuration = set_up_power_bin(&input);
124 
125  if (power_bin_configuration == NULL) {
126  if (input.file_path != NULL) {
127  acc_os_mem_free(input.file_path);
128  }
129  return EXIT_FAILURE;
130  }
131 
132  service_status = execute_power_bin(power_bin_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
133 
134  if (input.file_path != NULL) {
135  acc_os_mem_free(input.file_path);
136  }
137 
138  if (service_status != ACC_SERVICE_STATUS_OK) {
139  printf("execute_power_bin() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
140  return EXIT_FAILURE;
141  }
142 
143  acc_service_power_bins_configuration_destroy(&power_bin_configuration);
144  break;
145  }
146 
147  case ENVELOPE:
148  {
149  acc_service_configuration_t envelope_configuration = set_up_envelope(&input);
150 
151  if (envelope_configuration == NULL) {
152  if (input.file_path != NULL) {
153  acc_os_mem_free(input.file_path);
154  }
155  return EXIT_FAILURE;
156  }
157 
158  service_status = execute_envelope(envelope_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
159 
160  if (input.file_path != NULL) {
161  acc_os_mem_free(input.file_path);
162  }
163 
164  if (service_status != ACC_SERVICE_STATUS_OK) {
165  printf("execute_envelope() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
166  return EXIT_FAILURE;
167  }
168 
169  acc_service_envelope_configuration_destroy(&envelope_configuration);
170  break;
171  }
172 
173  case IQ:
174  {
175 
176  acc_service_configuration_t iq_configuration = set_up_iq(&input);
177 
178  if (iq_configuration == NULL) {
179  if (input.file_path != NULL) {
180  acc_os_mem_free(input.file_path);
181  }
182  }
183 
184  service_status = execute_iq(iq_configuration, input.file_path, input.wait_for_interrupt, input.sweep_count);
185 
186  if (input.file_path != NULL) {
187  acc_os_mem_free(input.file_path);
188  }
189 
190  if (service_status != ACC_SERVICE_STATUS_OK) {
191  printf("execute_iq() => (%u) %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
192  return EXIT_FAILURE;
193  }
194 
195  acc_service_iq_configuration_destroy(&iq_configuration);
196  break;
197  }
198  default:
199  {
200  printf("Invalid service_type %d\n", input.service_type);
201  return EXIT_FAILURE;
202  }
203  }
204 
205  acc_rss_deactivate();
206 
207  return EXIT_SUCCESS;
208 }

236 bool parse_options(int argc, char *argv[], input_t *input)
237 {
238  static struct option long_options[] =
239  {
240  {"service-type", required_argument, 0, 't'},
241  {"sweep-count", required_argument, 0, 'c'},
242  {"range-start", required_argument, 0, 'b'},
243  {"range-end", required_argument, 0, 'e'},
244  {"frequency", required_argument, 0, 'f'},
245  {"gain", required_argument, 0, 'g'},
246  {"number-of-bins", required_argument, 0, 'n'},
247  {"out", required_argument, 0, 'o'},
248  {"profile", required_argument, 0, 'p'},
249  {"running-avg-factor", required_argument, 0, 'r'},
250  {"sensor", required_argument, 0, 's'},
251  {"verbose", no_argument, 0, 'v'},
252  {"help", no_argument, 0, 'h'},
253  {NULL, 0, NULL, 0}
254  };
255 
256  int16_t character_code;
257  int32_t option_index = 0;
258 
259  while ((character_code = getopt_long(argc, argv, "t:c:b:e:f:g:n:o:p:r:s:vh?", long_options, &option_index)) != -1) {
260  switch (character_code) {
261  case 't':
262  {
263  switch (atoi(optarg)) {
264  case 0:
265  {
266  input->service_type = POWER_BIN;
267  break;
268  }
269  case 1:
270  {
271  input->service_type = ENVELOPE;
272  break;
273  }
274  case 2:
275  {
276  input->service_type = IQ;
277  break;
278  }
279  default:
280  printf("Invalid service type.\n");
281  print_usage();
282  return ACC_STATUS_FAILURE;
283  }
284  break;
285  }
286  case 'c':
287  {
288  input->sweep_count = atoi(optarg);
289  input->wait_for_interrupt = false;
290  break;
291  }
292  case 'b':
293  {
294  char *next;
295  input->start_m = strtof(optarg, &next);
296  break;
297  }
298  case 'e':
299  {
300  char *next;
301  input->end_m = strtof(optarg, &next);
302  break;
303  }
304  case 'f':
305  {
306  char *next;
307  float f = strtof(optarg, &next);
308  if (f > 0 && f < 100000) {
309  input->frequency = f;
310  }
311  else {
312  printf("Frequency out of range.\n");
313  print_usage();
314  exit(EXIT_FAILURE);
315  }
316  break;
317  }
318  case 'g':
319  {
320  char *next;
321  float g = strtof(optarg, &next);
322  if (g >= 0 && g <= 1) {
323  input->gain = g;
324  }
325  else {
326  printf("Gain out of range.\n");
327  print_usage();
328  exit(EXIT_FAILURE);
329  }
330  break;
331  }
332  case 'n':
333  {
334  int n = atoi(optarg);
335  if (n > 0 && n <= 32) {
336  input->n_bins = n;
337  }
338  else {
339  printf("Number of bins out of range.\n");
340  print_usage();
341  exit(EXIT_FAILURE);
342  }
343  break;
344  }
345  case 'p':
346  {
347  switch (atoi(optarg)) {
348  case 0:
349  {
350  input->profile = ACC_SERVICE_ENVELOPE_PROFILE_MAXIMIZE_SNR;
351  break;
352  }
353  case 1:
354  {
355  input->profile = ACC_SERVICE_ENVELOPE_PROFILE_MAXIMIZE_DEPTH_RESOLUTION;
356  break;
357  }
358  default:
359  printf("Invalid profile.\n");
360  print_usage();
361  exit(EXIT_FAILURE);
362  }
363  break;
364  }
365  case 'o':
366  {
367  input->file_path = acc_os_mem_alloc(sizeof(char) * (strlen(optarg) + 1));
368  if (input->file_path == NULL) {
369  printf("Failed allocating memory\n");
370  return ACC_STATUS_FAILURE;
371  }
372  snprintf(input->file_path, strlen(optarg) + 1, "%s", optarg);
373  break;
374  }
375  case 'r':
376  {
377  char *next;
378  float r = strtof(optarg, &next);
379  if (r >= 0 && r <= 1) {
380  input->running_avg = r;
381  }
382  else {
383  printf("Running average factor out of range.\n");
384  print_usage();
385  exit(EXIT_FAILURE);
386  }
387  break;
388  }
389  case 's':
390  {
391  int s = atoi(optarg);
392  if (s > 0 && s <= 4) {
393  input->sensor = s;
394  }
395  else {
396  printf("Sensor id out of range.\n");
397  print_usage();
398  exit(EXIT_FAILURE);
399  }
400  break;
401  }
402  case 'v':
403  {
404  acc_log_set_level(ACC_LOG_LEVEL_VERBOSE, NULL);
405  break;
406  }
407  case 'h':
408  case '?':
409  {
410  print_usage();
411  return ACC_STATUS_FAILURE;
412  }
413  }
414  }
415 
416  if (input->service_type == INVALID_SERVICE) {
417  printf("Missing option service type.\n");
418  print_usage();
419  return ACC_STATUS_FAILURE;
420  }
421 
422  return ACC_STATUS_SUCCESS;
423 }

460 acc_service_status_t execute_power_bin(acc_service_configuration_t power_bin_configuration, char *file_path, bool wait_for_interrupt, uint16_t sweep_count)
461 {
462  acc_service_handle_t handle = acc_service_create(power_bin_configuration);
463 
464  if (handle == NULL) {
465  printf("acc_service_create failed\n");
466  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
467  }
468 
469  acc_service_power_bins_metadata_t power_bins_metadata;
470  acc_service_power_bins_get_metadata(handle, &power_bins_metadata);
471 
472  float power_bins_data[power_bins_metadata.actual_bin_count];
473 
474  acc_service_power_bins_result_info_t result_info;
475  acc_service_status_t service_status = acc_service_activate(handle);
476 
477  if (service_status == ACC_SERVICE_STATUS_OK) {
478  FILE *file = stdout;
479 
480  if (file_path != NULL) {
481  file = fopen(file_path, "w");
482 
483  if(file == NULL) {
484  printf("opening file failed\n");
485  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
486  }
487  }
488 
489  uint16_t sweeps = 0;
490 
491  while ((wait_for_interrupt && interrupted == 0) || sweeps < sweep_count) {
492  service_status = acc_service_power_bins_get_next(handle, power_bins_data, power_bins_metadata.actual_bin_count, &result_info);
493 
494  if (service_status == ACC_SERVICE_STATUS_OK) {
495  for (uint_fast16_t index = 0; index < power_bins_metadata.actual_bin_count; index++) {
496  fprintf(file, "%u\t", (unsigned int)(power_bins_data[index] + 0.5));
497  }
498  fprintf(file, "\n");
499 
500  if (file_path == NULL) {
501  fflush(stdout);
502  }
503  }
504  else {
505  printf("Power bin data not properly retrieved\n");
506  fflush(stdout);
507  return ACC_SERVICE_STATUS_FAILURE_UNSPECIFIED;
508  }
509 
510  if (!wait_for_interrupt) {
511  sweeps++;
512  }
513  }
514 
515  if (file_path != NULL) {
516  fclose(file);
517  }
518 
519  service_status = acc_service_deactivate(handle);
520  }
521  else {
522  printf("acc_service_activate() (%u) => %s\n", (unsigned int)service_status, acc_service_status_name_get(service_status));
523  }
524 
525  acc_service_destroy(&handle);
526 
527  return service_status;
528 }
